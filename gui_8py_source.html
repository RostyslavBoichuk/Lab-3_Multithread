109class ComparisonThread(ProcessingThread):
110
115 def __init__(self, image: OpenCVImage, operation: str, method_or_shift, num_runs: int = 3):
116 super().__init__()
117 self.image = image
118 self.operation = operation
119 self.method_or_shift = method_or_shift
120 self.num_runs = num_runs
121 self.results = {}
122
123 def run(self):
124 try:
125 if self.operation == 'monochrome':
126 self.compare_monochrome()
127 else:
128 self.compare_hue()
129 self.result_ready.emit(self.results)
130 except Exception as e:
131 self.error_occurred.emit(f"Error in comparison: {str(e)}")
132 finally:
133 self.finished.emit()
134
135 def compare_monochrome(self):
136 method = self.method_or_shift
137
138 seq_times = []
139 for _ in range(self.num_runs):
140 start = time.time()
141 converter = ProcessorFactory.create_monochrome_converter(
142 mode=ProcessingMode.SEQUENTIAL,
143 method=method
144 )
145 converter.process(self.image)
146 seq_times.append(time.time() - start)
147
148 par_times = {}
149 for num_threads in [1, 2, 4]:
150 times = []
151 for _ in range(self.num_runs):
152 start = time.time()
153 converter = ProcessorFactory.create_monochrome_converter(
154 mode=ProcessingMode.PARALLEL,
155 method=method,
156 num_threads=num_threads
157 )
158 converter.process(self.image)
159 times.append(time.time() - start)
160 par_times[num_threads] = times
161
162 self.results = {
163 'seq_avg': sum(seq_times) / len(seq_times),
164 'seq_times': seq_times,
165 'par_times': par_times
166 }
167
168 def compare_hue(self):
169 hue_shift = self.method_or_shift
170
171 seq_times = []
172 for _ in range(self.num_runs):
173 start = time.time()
174 adjuster = ProcessorFactory.create_hue_adjuster(
175 mode=ProcessingMode.SEQUENTIAL
176 )
177 adjuster.set_hue_shift(hue_shift)
178 adjuster.process(self.image)
179 seq_times.append(time.time() - start)
180
181 par_times = {}
182 for num_threads in [1, 2, 4]:
183 times = []
184 for _ in range(self.num_runs):
185 start = time.time()
186 adjuster = ProcessorFactory.create_hue_adjuster(
187 mode=ProcessingMode.PARALLEL,
188 num_threads=num_threads
189 )
190 adjuster.adjust_hue_batch([self.image], hue_shift)
191 times.append(time.time() - start)
192 par_times[num_threads] = times
193
194 self.results = {
195 'seq_avg': sum(seq_times) / len(seq_times),
196 'seq_times': seq_times,
197 'par_times': par_times
198 }

203class ImageViewer(QLabel):
204
205 def __init__(self):
206 super().__init__()
207 self.setMinimumSize(500, 400)
208 self.setStyleSheet("border: 2px solid #ccc; background-color: #f5f5f5;")
209 self.setAlignment(Qt.AlignCenter)
210 self.setText("No image loaded")
211 self.setScaledContents(False)

213 def display(self, cv_image):
214
216 if cv_image is None or cv_image.size == 0:
217 self.setText("Invalid image")
218 return
219
220 try:
221 rgb = cv2.cvtColor(cv_image, cv2.COLOR_BGR2RGB)
222
223 h, w = rgb.shape[:2]
224 max_dim = 500
225 if h > max_dim or w > max_dim:
226 scale = min(max_dim / h, max_dim / w)
227 rgb = cv2.resize(rgb, (int(w * scale), int(h * scale)))
228
229 h, w, ch = rgb.shape
230 bytes_per_line = 3 * w
231 qt_img = QImage(rgb.data, w, h, bytes_per_line, QImage.Format_RGB888)
232 pixmap = QPixmap.fromImage(qt_img)
233
234 self.setPixmap(pixmap)
235 except Exception as e:
236 self.setText(f"Error displaying image: {str(e)}")
